def validate_document(file_path, doc_type):
    """ë¬¸ì„œ í’ˆì§ˆ ìë™ ê²€ì¦
    
    Args:
        file_path (Path): ë¬¸ì„œ ê²½ë¡œ
        doc_type (str): 'pptx', 'pdf', 'xlsx', 'md'

        dict: {'passed': bool, 'issues': list}

    # ê³µí†µ ê²€ì¦
    if not file_path.exists():
        issues.append(f"[ì¹˜ëª…ì ] íŒŒì¼ ì—†ìŒ: {file_path}")

    file_size = file_path.stat().st_size

            issues.append("[ì˜¤ë¥˜] ìŠ¬ë¼ì´ë“œ ì—†ìŒ")

                issues.append(f"[ê²½ê³ ] ìŠ¬ë¼ì´ë“œ {i}: ì œëª© ì—†ìŒ")
    
    elif doc_type == 'pdf':
        from PyPDF2 import PdfReader
        reader = PdfReader(file_path)
        
        if len(reader.pages) == 0:
            issues.append("[ì˜¤ë¥˜] í˜ì´ì§€ ì—†ìŒ")
        
        # PDF/A ì¤€ìˆ˜ í™•ì¸

            issues.append("[ì˜¤ë¥˜] ì‹œíŠ¸ ì—†ìŒ")

    passed = not any('[ì¹˜ëª…ì ]' in i or '[ì˜¤ë¥˜]' in i for i in issues)
    return {'passed': passed, 'issues': issues}
