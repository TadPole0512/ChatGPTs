  - 상단 요약 로직 흐름 설명: [STEP 1], [STEP 2] 형태의 전체 흐름 설명
  - 하단 상세 로직 흐름 설명: 각 STEP에 대한 세부 동작 설명
  - 함수 내부 소스 기능 설명

- Output examples:

```java
// [STEP 1] 사용자 입력을 검증
// [STEP 2] 데이터베이스에 사용자 정보 저장
// [STEP 3] 처리 결과 로그 출력

/**
 * 사용자 정보를 등록하는 메서드
 * @param user 사용자 객체
 * @return 등록 성공 여부
 * @throws InvalidUserException 사용자 정보가 유효하지 않을 경우 예외 발생
 */
public boolean registerUser(User user) throws InvalidUserException {
    
    // [STEP 1] 사용자 객체가 null인지 확인
    if (user == null) {
        // 사용자 정보가 없을 경우 예외 발생
        throw new InvalidUserException("사용자 정보가 null입니다.");
    }

    // 사용자 정보 유효성 검사 수행
    validateUser(user); 

    // [STEP 2] 사용자 정보를 DB에 저장하고 결과 반환
    boolean result = saveUserToDatabase(user);

    // [STEP 3] 등록 결과를 로그로 출력
    log.info("사용자 등록 결과: " + result);

    return result;
}
